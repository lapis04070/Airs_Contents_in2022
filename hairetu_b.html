 float[] x = new float[1000];図形を表示する数

void setup(){
size(240,120); //ウインドウを描く
noStroke();
fill(255, 200);
for(int i = 0; i < x.length; i++){
x[i] = random(-500, 50); //x座標が-500から50の中でランダムで作成した図形が表示される
}
}

void draw(){
background(0);
for(int i = 0; i < x.length; i++){
x[i] += 0.5;
float y = i * 0.4;
rect(x[i], y, 10, 10);表示する図形
}
}

実行結果
変数で1000個のコードを1つずつ追加するより、配列を使うことでコードがすっきりになりました。
