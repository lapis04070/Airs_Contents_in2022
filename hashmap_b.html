 class Chicken{
int x;
int y;
String name;
Chicken(int tempX,int tempY){
x = tempX;
y = tempY;
}

void display(){
pushMatrix();
translate(x,y);
stroke(255);
strokeWeight(50);
line(0,-35,0,-65);
noStroke();
fill(255);
arc(0, -30, 70, 70, PI, TWO_PI);
fill(51, 51, 30);
ellipse(-15, -60, 8, 8);
ellipse( 15, -60, 8, 8);
fill(255, 255, 0);
quad(0, -65, 15, -51, 0, -45, -15, -51);
fill(255, 0, 0);
quad(0, -105, 5, -90, 0, -80, -5, -90);
quad(0, -45, 5, -30, 0, -20, -5, -30);
popMatrix();
}
}

 HashMap hashMap = new HashMap();

void setup(){
size(480, 120);
