複数のオブジェクトを管理する際に、あらかじめ要素の数を決めておく必要がありますが、この問題を解決するオブジェクトがArrayListオブジェクトです。

 ArrayListarrayList = new ArrayList();

void setup(){
size(770,120); //ウインドウを描く
background(176, 204, 226);
for(int i = 0; i < 11; i++){ //表示する数を指定
if((i % 2) == 0){ //iを2で割った時の余りが0か1で判断できる。
arrayList.add(new Owl(35 + i * 70, 110));
} else { //余りが0ではない場合は、下のChickenが実行
arrayList.add(new Chicken(35 + i * 70, 110));
}
}
}
void draw(){
for(Bird bird:arrayList){
bird.display();

}
