オブジェクト Bパターン

複数のオブジェクト生成

HashMapオブジェクトは、Processingで標準的に利用できるので、HashMapクラスを定義する必要はありません。

今回は、チキン(Chicken)とフクロウ(Owl)の2つのオブジェクトを生成します。

step.1

始めに表示するチキン(Chicken)とフクロウ(Owl)のクラスを定義します。
タブ名は「Chicken」と「Owl」にします。

 class Chicken{
int x;
int y;
Chicken(int tempX,int tempY){
x = tempX;
y = tempY;
}

void display(){
pushMatrix();
translate(x,y);
stroke(255);
strokeWeight(50);
line(0,-35,0,-65);
noStroke();
fill(255);
arc(0, -30, 70, 70, PI, TWO_PI);
fill(51, 51, 30);
ellipse(-15, -60, 8, 8);
ellipse( 15, -60, 8, 8);
fill(255, 255, 0);
quad(0, -65, 15, -51, 0, -45, -15, -51);
fill(255, 0, 0);
quad(0, -105, 5, -90, 0, -80, -5, -90);
quad(0, -45, 5, -30, 0, -20, -5, -30);
popMatrix();
}
}

 class Owl{
int x;
int y;
Owl(int tempX,int tempY){
x = tempX;
y = tempY;
}

void display(){
pushMatrix();
translate(x,y);
stroke(138,138,125);
strokeWeight(70);
line(0,-35,0,-65);
noStroke();
fill(255);
ellipse(-17.5,-65,35,35);
ellipse(17.5,-65,35,35);
arc(0,-65,70,70,0,PI);
fill(51,51,30);
ellipse(-14,-65,8,8);
ellipse(14,-65,8,8);
quad(0,-58,4,-51,0,-44,-4,-51);
popMatrix();
}
}

 Chicken chicken;//オブジェクトを宣言
Owl owl;

void setup(){
size(480,120);
chicken = new Chicken(40,110);
owl = new Owl(440,110);
}

void draw(){
chicken.display();
owl.display();
}
